function buildRoundFeedback(m,allPrev){
  const signals=[];
  if(m.timeout)signals.push({txt:"Tempo esgotado — resposta incompleta",cls:"tr"});
  if(!m.response||m.response==="(sem resposta)")signals.push({txt:"Sem resposta enviada",cls:"tr"});
  if(m.wc<8)signals.push({txt:"Resposta muito curta",cls:"tr"});
  if(m.wc>90)signals.push({txt:"Resposta muito longa — pode ser conteúdo copiado",cls:"ta"});
  if(m.wc>=12&&m.wc<=70)signals.push({txt:"Extensão adequada",cls:"tg"});
  if(/\?/.test(m.response))signals.push({txt:"Fez perguntas ao cliente",cls:"tg"});
  if(m.elapsed<4&&m.wc>20)signals.push({txt:"Resposta muito rápida para o tamanho — possível cópia",cls:"ta"});
  if(allPrev.length>=2){
    const prev=allPrev[allPrev.length-2];
    const lexSim=lexicalSimilarity(m.response,prev.response);
    if(lexSim>0.55)signals.push({txt:"Alta similaridade com resposta anterior",cls:"tr"});
    else if(lexSim<0.25)signals.push({txt:"Abordagem diferente da rodada anterior",cls:"tg"});
  }
  return signals;
}

function lexicalSimilarity(a,b){
  if(!a||!b)return 0;
  const setA=new Set(a.toLowerCase().split(/\W+/).filter(w=>w.length>3));
  const setB=new Set(b.toLowerCase().split(/\W+/).filter(w=>w.length>3));
  if(!setA.size||!setB.size)return 0;
  let inter=0;setA.forEach(w=>{if(setB.has(w))inter++;});
  return inter/Math.max(setA.size,setB.size);
}

function clusterProfileSimilarity(pa,pb){
  let match=0,total=0;
  ARG_CLUSTERS.forEach(cl=>{
    const a=Math.min(1,pa[cl.name]||0),b=Math.min(1,pb[cl.name]||0);
    if(a||b){total++;if(a&&b)match++;}
  });
  return total?match/total:0;
}

function buildFullAnalysis(resps){
  let persist=50,adapt=50,comm=50,emoc=50;
  const flags=[];
  const consistencyBlocks=[];
  const noResp=resps.filter(r=>!r.response||r.response==="(sem resposta)"||r.wc<5).length;
  const timeouts=resps.filter(r=>r.timeout).length;
  persist-=noResp*12;persist-=timeouts*8;
  if(noResp===0&&timeouts===0)persist+=15;
  if(resps.filter(r=>r.wc>=8).length===5)persist+=10;
  resps.forEach(r=>{if(r.wc>=8)persist+=3;});
  const argProfiles=resps.map(r=>getArgProfile(r.response));
  let argDiversity=0;
  for(let i=1;i<argProfiles.length;i++){
    const sim=clusterProfileSimilarity(argProfiles[i],argProfiles[i-1]);
    argDiversity+=(1-sim);
    if(sim>0.65){adapt-=8;flags.push({text:`Rodadas ${i} e ${i+1}: mesmo argumento central`,positive:false});}
    else if(sim<0.25){adapt+=8;flags.push({text:`Rodadas ${i} e ${i+1}: abordagem diferente detectada`,positive:true});}
  }
  const avgDiv=argDiversity/(argProfiles.length-1||1);
  if(avgDiv>0.6)adapt+=12;
  let lexVar=0;
  for(let i=1;i<resps.length;i++){
    const ls=lexicalSimilarity(resps[i].response,resps[i-1].response);
    if(ls>0.55){adapt-=6;}else lexVar++;
  }
  if(lexVar>=3)adapt+=8;
  const questionRounds=resps.filter(r=>/\?/.test(r.response)).length;
  adapt+=questionRounds*4;
  if(questionRounds>=3)flags.push({text:"Fez perguntas ao cliente em "+questionRounds+" rodadas",positive:true});
  const pressaResp=resps.find(r=>r.context==="pressa");
  if(pressaResp){
    if(pressaResp.wc>60){adapt-=10;flags.push({text:"Cliente com pressa: resposta longa demais ("+pressaResp.wc+" palavras)",positive:false});}
    else if(pressaResp.wc<=30&&pressaResp.wc>=5){adapt+=8;flags.push({text:"Adaptou o tamanho da resposta para cliente com pressa",positive:true});}
  }
  const wcArr=resps.map(r=>r.wc);
  const avgWc=wcArr.reduce((a,b)=>a+b,0)/wcArr.length;
  if(avgWc<8)comm-=20;
  else if(avgWc>=12&&avgWc<=55)comm+=15;
  else if(avgWc>80)comm-=10;
  const allText=resps.map(r=>r.response).join(" ").toLowerCase();
  const allWords=allText.split(/\W+/).filter(w=>w.length>3);
  const uniqueRatio=allWords.length>0?new Set(allWords).size/allWords.length:0;
  if(uniqueRatio<0.35){comm-=12;flags.push({text:"Vocabulário muito repetitivo ao longo do teste",positive:false});}
  else if(uniqueRatio>0.55)comm+=8;
  const wcTrend=resps.slice(1).reduce((acc,r,i)=>acc+(r.wc>resps[i].wc?1:-1),0);
  if(wcTrend>=2){comm+=5;flags.push({text:"Engajamento crescente ao longo das rodadas",positive:true});}
  else if(wcTrend<=-3){comm-=8;flags.push({text:"Desengajamento progressivo detectado",positive:false});}
  const timeUsedAvg=resps.reduce((a,r)=>a+r.timeUsedPct,0)/resps.length;
  if(timeouts>=3){emoc-=20;flags.push({text:timeouts+" rodadas com tempo esgotado",positive:false});}
  else if(timeouts===0){emoc+=12;}
  if(timeUsedAvg>0.2&&timeUsedAvg<0.9)emoc+=8;
  if(timeUsedAvg<0.08){emoc-=8;flags.push({text:"Respostas muito rápidas — possível conteúdo pré-preparado",positive:false});}
  const allLong=resps.filter(r=>r.wc>80).length;
  if(allLong>=3){adapt-=10;comm-=8;flags.push({text:allLong+" respostas muito longas — padrão de conteúdo copiado",positive:false});}
  for(let i=0;i<resps.length;i++){
    for(let j=i+1;j<resps.length;j++){
      const sim=lexicalSimilarity(resps[i].response,resps[j].response);
      if(sim>0.75&&resps[i].wc>10){adapt-=15;flags.push({text:`Respostas das rodadas ${i+1} e ${j+1} quase idênticas`,positive:false});}
    }
  }
  const argUsage={};
  ARG_CLUSTERS.forEach(cl=>{argUsage[cl.name]=0;});
  resps.forEach(r=>{const p=getArgProfile(r.response);ARG_CLUSTERS.forEach(cl=>{if(p[cl.name]>0)argUsage[cl.name]++;});});
  const dominant=Object.entries(argUsage).sort((a,b)=>b[1]-a[1])[0];
  if(dominant[1]>=4){consistencyBlocks.push({title:"Argumento dominante",text:`Usou "${dominant[0].replace("_"," ")}" em ${dominant[1]} de 5 rodadas — baixa diversidade estratégica.`});adapt-=8;}
  else{consistencyBlocks.push({title:"Diversidade de argumentos",text:"Variou estratégia ao longo das rodadas — boa adaptação detectada."});}
  const hasPergunta=resps.filter(r=>/\?/.test(r.response)).length;
  consistencyBlocks.push({title:"Tentativa de entender o cliente",text:hasPergunta>=3?"Fez perguntas em "+hasPergunta+" rodadas — demonstra orientação ao cliente.":hasPergunta>=1?"Fez perguntas em "+hasPergunta+" rodada(s) — pode desenvolver mais.":"Nunca perguntou ao cliente — postura mais passiva."});
  const evolucao=wcTrend>=1?"Respostas foram crescendo — sinal de engajamento.":wcTrend<=-2?"Respostas foram diminuindo — possível desistência progressiva.":"Nível de engajamento estável ao longo do desafio.";
  consistencyBlocks.push({title:"Evolução de engajamento",text:evolucao});
  const clamp=(v)=>Math.max(0,Math.min(100,Math.round(v)));
  const pFinal=clamp(persist),aFinal=clamp(adapt),cFinal=clamp(comm),eFinal=clamp(emoc);
  const total=Math.round((pFinal+aFinal+cFinal+eFinal)/4);
  return{scores:{persist:pFinal,adapt:aFinal,comm:cFinal,emoc:eFinal,total},flags,consistencyBlocks,profile:buildProfile(pFinal,aFinal,cFinal,eFinal)};
}

function buildProfile(p,a,c,e){
  const hi=v=>v>=65,lo=v=>v<42;
  if(hi(p)&&hi(a)&&hi(c))return "Vendedor completo — persistente, adaptável e comunicativo";
  if(hi(p)&&hi(a))return "Vendedor persistente e adaptável — alto potencial";
  if(hi(p)&&lo(a))return "Persistente mas repetitivo — precisa variar a abordagem";
  if(hi(a)&&lo(p))return "Adaptável mas cede rápido — trabalhar a persistência";
  if(hi(c)&&lo(a))return "Bom comunicador, pouca adaptação estratégica";
  if(hi(e)&&lo(c))return "Resiliente sob pressão, comunicação a desenvolver";
  if(lo(p)&&lo(a))return "Iniciante — desenvolver persistência e adaptação";
  const avg=(p+a+c+e)/4;
  if(avg>=65)return "Perfil equilibrado — bom potencial";
  if(avg>=45)return "Perfil em desenvolvimento — base sólida";
  return "Perfil iniciante — potencial a desenvolver com prática";
}
